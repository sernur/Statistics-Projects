  Usim=rnorm(50)*S
  Ysim= X%*%fit$coef + Usim
  fitsim=lm(Ysim~x1+x2)
  b2sim[i,]=fitsim$coef

  b1sim[i,]=parsim$par
  error1=error1+sum((b1sim[i,]-fit$coef)^2)
  error2=error2+sum((b2sim[i,]-fit$coef)^2)
}
